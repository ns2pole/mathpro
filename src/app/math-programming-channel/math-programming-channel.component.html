    elem' :: Eq a => a -> [a] -> Bool
    elem' _ [] = False
    elem' a (x:xs)
      | a == x = True
      | otherwise = elem a xs
    bool1 = elem' 1 [1,2,3]
    bool2 = elem' 1 [2,3,4]
    main :: IO ()
    main = do
      putStrLn $ show (bool1)
