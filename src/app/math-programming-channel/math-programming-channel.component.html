    elem' :: Eq a => a -> [a] -> Bool
    elem' _ [] = False
    elem' a (x:xs)
      | a == x = True
      | otherwise = elem a xs
    bool1 = elem' 1 [1,2,3]
    bool2 = elem' 1 [2,3,4]
    main :: IO ()
    main = do
      putStrLn $ show (bool1)
      putStrLn $ show (bool2)

    insert :: Int -> Int -> [Int] -> [Int]
    insert index x list = (take index list) ++ [x] ++ (drop index list)
    f i= \l -> insert i (l !! i) $ filter (\a -> a `mod` (l !! i) /= 0) l
    fs = foldl1 (.) $ reverse $ map f [0..100]
    pl = fs [2..1000]
    pf :: Int -> [Int] -> [Int]
    pf x (p:ps)
      | x == 1 = []
      | x `mod` p == 0 = p : pf (x `div` p) (p:ps)
      | otherwise = pf x ps
