    elem' _ [] = False

      | a == x = True
      | otherwise = elem a xs

    insert :: Int -> Int -> [Int] -> [Int]
    insert index x list = (take index list) ++ [x] ++ (drop index list)
    f i= \l -> insert i (l !! i) $ filter (\a -> a `mod` (l !! i) /= 0) l
    fs = foldl1 (.) $ reverse $ map f [0..100]
    pl = fs [2..1000]
    pf :: Int -> [Int] -> [Int]
    pf x (p:ps)
      | x == 1 = []
      | x `mod` p == 0 = p : pf (x `div` p) (p:ps)
      | otherwise = pf x ps
    main = putStrLn (show $ pf (124124 ::Int) pl)

    root :: Integer -> Double
    root x = sqrt (fromIntegral x)
    f :: Integer -> [(Integer, Integer)]
    f x = [(a, x `div` a) | a <- [1..(floor (root x))], x `mod` a == 0]
